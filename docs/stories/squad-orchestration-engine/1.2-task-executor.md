**so that** posso processar steps de um pipeline.

---

## Description

Este story implementa o TaskExecutor, componente responsável por executar os steps definidos em um pipeline. O TaskExecutor deve:

1. Carregar task file de `squads/{name}/tasks/{task}.md`
2. Executar **task_pura** (função JavaScript mapeada ou script)
3. Executar **agent_task** (invocar AgentInvoker com agente especificado)
4. Passar input definido no step com interpolação de variáveis (`{{context.field}}`)
5. Retornar output estruturado conforme task file
6. Salvar output no context acumulado do run

**Note:** Event emission (step.completed) deferred to Story 1.7 (Event Sourcing)

**Contexto Técnico:**

O TaskExecutor é o coração da execução do pipeline. Ele processa cada step sequencialmente, resolvendo dependências de contexto e invocando tasks ou agentes conforme necessário.

**Integração com Sistema Existente:**

- Reutiliza `AgentInvoker` existente para agent tasks
- Integra com `SquadOrchestrator` (Story 1.1) para carregar playbooks
- Novo componente: `.aios-core/core/orchestration/squad-engine/task-executor.js`

### TaskExecutor Class Outline

```javascript
// .aios-core/core/orchestration/squad-engine/task-executor.js

const fs = require('fs').promises;
const path = require('path');
const { AgentInvoker } = require('../agent-invoker');
const PureTaskRunner = require('./task-types/pure-task-runner');
const AgentTaskRunner = require('./task-types/agent-task-runner');

class TaskExecutor {
  constructor() {
    this.pureTaskRunner = new PureTaskRunner();
    this.agentTaskRunner = new AgentTaskRunner();
  }

  /**
   * Executes a task step from pipeline
   * @param {Object} step - Step definition from pipeline.yaml
   * @param {Object} context - Current run context
   * @param {string} squadName - Name of squad
   * @returns {Promise<Object>} Task output with execution metadata
   */
  async executeTask(step, context, squadName) {
    const startTime = Date.now();

    // 1. Load task file
    const taskFile = await this.loadTaskFile(squadName, step.task);

    // 2. Interpolate input with context
    const interpolatedInput = this.interpolateInput(step.input, context);

    // 3. Execute based on task type
    let output;
    if (step.type === 'task_pura') {
      output = await this.pureTaskRunner.execute(taskFile, interpolatedInput);
    } else if (step.type === 'agent_task') {
      output = await this.agentTaskRunner.execute(
        step.agent,
        step.task,
        interpolatedInput
      );
    } else {
      throw new Error(`Unknown task type: ${step.type}`);
    }

    // 4. Add execution metadata
    const duration = Date.now() - startTime;

    // NOTE: Event emission deferred to Story 1.7 (Event Sourcing)
    // When EventStore is available, add:
    // this.eventStore.append({ event: 'step.completed', ... })

    return {
      output,
      metadata: {
        stepId: step.id,
        task: step.task,
        duration,
        completedAt: new Date().toISOString(),
      },
    };
  }

  /**
   * Loads task file from squad directory
   * @param {string} squadName
   * @param {string} taskName
   * @returns {Promise<Object>} Parsed task definition
   */
  async loadTaskFile(squadName, taskName) {
    const taskPath = path.join(
      process.cwd(),
      'squads',
      squadName,
      'tasks',
      `${taskName}.md`
    );

    const content = await fs.readFile(taskPath, 'utf8');

    // Parse frontmatter and extract task definition
    // (Simplified - full implementation would use a proper markdown parser)
    return {
      name: taskName,
      content,
    };
  }

  /**
   * Interpolates variables in input using context
   * @param {any} input - Input value (string, object, array)
   * @param {Object} context - Run context
   * @returns {any} Interpolated input
   */
  interpolateInput(input, context) {
    if (typeof input === 'string') {
      return input.replace(/\{\{([^}]+)\}\}/g, (match, path) => {
        return this.resolveContextPath(path.trim(), context);
      });
    }

    if (Array.isArray(input)) {
      return input.map(item => this.interpolateInput(item, context));
    }

    if (typeof input === 'object' && input !== null) {
      const result = {};
      for (const [key, value] of Object.entries(input)) {
        result[key] = this.interpolateInput(value, context);
      }
      return result;
    }

    return input;
  }

  /**
   * Resolves a path in context (e.g., "step1.output.value")
   * @param {string} path - Dot-notation path
   * @param {Object} context - Run context
   * @returns {any} Resolved value
   */
  resolveContextPath(path, context) {
    const parts = path.split('.');
    let current = context;

    for (const part of parts) {
      if (current === undefined || current === null) {
        return undefined;
      }
      current = current[part];
    }

    return current;
  }
}

module.exports = TaskExecutor;
```

const { spawn } = require('child_process');

class PureTaskRunner {
  /**
   * Executes a pure task (JavaScript function or script)
   * @param {Object} taskFile - Task definition
   * @param {Object} input - Interpolated input
   * @returns {Promise<Object>} Task output
   */
  async execute(taskFile, input) {
    // For MVP: support basic JavaScript execution
    // Future: support Python, shell scripts, etc.

    // Example: if task is mapped to a function
    const taskFn = this.getTaskFunction(taskFile.name);

    if (taskFn) {
      return await taskFn(input);
    }

    throw new Error(`Task function not found: ${taskFile.name}`);
  }

  /**
   * Maps task name to implementation function
   * @param {string} taskName
   * @returns {Function|null}
   */
  getTaskFunction(taskName) {
    const taskMap = {
      'fetch-offer-data': this.fetchOfferData.bind(this),
      // Add more task mappings here
    };

    return taskMap[taskName] || null;
  }

  /**
   * Example: fetch-offer-data task implementation
   */
  async fetchOfferData(input) {
    const fs = require('fs').promises;
    const yaml = require('js-yaml');
    const path = require('path');

    const offerPath = path.join(
      process.cwd(),
      'data',
      'offers',
      input.offerId,
      'offer.yaml'
    );

    const content = await fs.readFile(offerPath, 'utf8');
    const offer_context = yaml.load(content);

    return { offer_context };
  }
}

module.exports = PureTaskRunner;

### AgentTaskRunner Implementation

```javascript
// .aios-core/core/orchestration/squad-engine/task-types/agent-task-runner.js

const { AgentInvoker } = require('../../agent-invoker');

class AgentTaskRunner {
  constructor() {
    this.agentInvoker = new AgentInvoker();
  }

  /**
   * Executes an agent task using existing AgentInvoker
   * @param {string} agent - Agent name (e.g., "@copy-chief")
   * @param {string} task - Task name
   * @param {Object} input - Interpolated input
   * @returns {Promise<Object>} Agent output
   */
  async execute(agent, task, input) {
    // Use correct AgentInvoker API (Story 1.1 blocker fix)
    const result = await this.agentInvoker.invokeAgent(
      agent,      // agentName
      task,       // taskPath
      input       // inputs
    );

    // Handle invocation errors
    if (!result.success) {
      throw new Error(`Agent invocation failed: ${result.error}`);
    }

    // Return agent output
    return result.result;
  }
}

module.exports = AgentTaskRunner;
```
