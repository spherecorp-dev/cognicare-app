function_create_vsl:

  description: "Gerar VSL script do zero, adaptado para diferentes duracoes"

  process_long_vsl:
    name: "VSL Longa (15-45 min)"
    steps:
      - step: 1
        name: "Pesquisa (Fact Sheet)"
        action: |
          Organizar TODAS as informacoes em:
          - Fact Sheet: cada fato sobre o produto/oferta
          - Benefit List: cada fato traduzido em beneficio direto
          - Pain Points: dores especificas da Starving Crowd
          - Proof Inventory: todas as provas disponiveis

      - step: 2
        name: "Estrutura AIDA"
        action: |
          Mapear o fluxo da VSL na estrutura AIDA:
          - Atencao: Hook/Grabber
          - Interesse: Story + Problem Amplification
          - Desejo: Mechanism + Proof Stacking
          - Acao: Offer + Close

      - step: 3
        name: "Script Bloco a Bloco"
        blocks:

          hook:
            duration: "0-30s"
            technique: "IF/THEN de Halbert ou Pattern Interrupt agressivo"
            description: |
              O grabber DECIDE TUDO. Deve ser impossivel de ignorar.
              Teste A-pile: se aparecesse na sua caixa de correio,
              voce abriria ou jogaria fora?
            formulas:
              if_then: "Se voce [situacao do viewer], entao os proximos [tempo] vao mudar sua vida..."
              pattern_interrupt: "[Afirmacao chocante/contraintuitiva que quebra o scroll]"
              thirteen_word_opener: "Abertura em 13 palavras ou menos que captura atencao instantanea"
              question: "Voce sabia que [fato chocante e especifico]?"
            rule: "Escrever 20+ variantes de grabber. Selecionar os 3 melhores pra teste."

          story:
            duration: "30s-5min"
            technique: "Epiphany Bridge (Brunson) com vulnerabilidade Halbert"
            description: |
              Historia pessoal que conecta a dor do publico com a descoberta da solucao.
              Vulnerabilidade REAL — mostrar fracasso, vergonha, desespero.
              O viewer deve pensar: "Essa pessoa e EXATAMENTE como eu."

          agitate_and_educate:
            duration: "5-12min"
            technique: "Agitacao Halbert + Mecanismo Georgi"
            description: |
              Agitar a dor com DETALHES VIVIDOS e sensoriais.
              Depois educar sobre a CAUSA RAIZ do problema.
              Introduzir o Mecanismo Unico como a "nova descoberta".
              O mecanismo deve ser: novo, plausivel, proprietario, nomeavel.
            agitation_rules:
              - "Cenarios especificos: 'Voce sabe aquele momento quando...'"
              - "Linguagem sensorial: vista, tato, emocao"
              - "Invalidar solucoes anteriores: 'Nao e culpa sua que X nao funcionou...'"
            mechanism_rules:
              - "Nomear o mecanismo (nome proprietario que gruda)"
              - "Explicar POR QUE nada funcionou antes"
              - "Credencializar com fonte (estudo, medico, universidade)"
              - "Manter compliance: focar no mecanismo, nao em claims"

          proof:
            duration: "12-18min"
            technique: "Proof Stacking com especificidade Halbert"
            description: |
              Empilhar provas de tipos DIFERENTES ate que duvida seja impossivel:
              1. Testimonial pessoal (historia do criador)
              2. Social proof com NUMEROS ("1.726 clientes em 32 estados")
              3. Expert endorsement (Dr./Prof. fulano recomenda)
              4. Dados/estudos (pesquisa da Universidade X)
              5. Demonstracao logica (faz sentido porque...)
              6. Antes/depois (transformacao especifica)
              7. Testimonials de terceiros (3-5 historias curtas COM NOMES)
            rules:
              - "MINIMO 3 tipos diferentes de prova"
              - "NUNCA 'muitos clientes'. SEMPRE '1.726 clientes em 32 estados'"
              - "Especificidade = credibilidade"

          offer_and_close:
            duration: "18-25min"
            technique: "The Stack (Brunson) + Reason Why (Halbert)"
            description: |
              Apresentar a oferta como CONCLUSAO NATURAL.
              Usar The Stack para empilhar valor (5-10x preco).
              Adicionar garantia FORTE (risk reversal).
              Criar urgencia com REASON WHY real (nao fabricada).
              Fechar ocupando pelo menos 20% do script.
            close_structure:
              - "Recapitular os 3 maiores beneficios"
              - "Custo de NAO agir (dor de ficar onde esta)"
              - "The Stack com valores individuais"
              - "Garantia nomeada e ousada"
              - "Urgencia com Reason Why: POR QUE esse preco, POR QUE agora"
              - "CTA repetido minimo 3 vezes"
              - "P.S. com beneficio mais forte (segundo mais lido/ouvido)"

  process_mini_vsl:
    name: "Mini VSL (2-5 min)"
    structure:
      hook: {duration: "5s", content: "Problema + Mecanismo comprimido"}
      agitation: {duration: "30s", content: "Dor vivida e rapida"}
      solution: {duration: "60s", content: "Mecanismo Unico apresentado"}
      proof: {duration: "30s", content: "1 testimonial ou prova social rapida"}
      offer: {duration: "30s", content: "CTA direto e claro"}

      detail: "Elipses criam pausa... mantem o viewer em movimento... simulam conversa."
    - rule: "Ritmo variado"
      detail: "Frases curtas. Depois longas. Depois curtas de novo. Monotonia = abandono."
    - rule: "Use sublinhado, CAPS e **negrito** com intencao"

    - "Mas aqui e que fica REALMENTE interessante..."
    - "E isso nao e tudo..."
    - "Agora, preste muita atencao..."
    - "Voce vai entender por que isso e tao importante em um segundo..."
    - "Mas antes de eu te mostrar, preciso te avisar..."
    - "E tem mais..."
    - "Agora vem a parte que muda tudo..."
    - "Pare e pense nisso por um segundo..."

    - "Linguagem passiva (voz passiva = morte do VSL)"
    - "Paragrafos com mais de 3 linhas"
    - "Frases sem proposito (se nao vende, DELETE)"

    - "Generico em vez de especifico"
